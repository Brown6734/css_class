    <!--java script below-->

    <script>
        //Javascript here

        //variable for users hourly pay
        //hourlyPay = prompt(`prompt`)
        hourlyPay = Number(prompt(`What is your hourly pay?: `))

        //variable for hours worked in a two week period
        hoursWorked = Number(prompt(`How many hours do you work in a two week period: `))

        //grossPay = hourly pay * hours worked
        grossPay = hourlyPay * hoursWorked

        //variable for tax rate; use 20%
        //taxRate = 20/100
        taxRate = 20/100

        //tax amount = gross pay * tax rate
        taxAmount = grossPay * taxRate

        //net pay = gross pay - tax amount
        netPay = grossPay - taxAmount

        //display the user's Gross Pay, Uncle Sam's Share (taxes), and the User's Net pay

        //use one alert() function as well as three console.log() calls to display this info

        console.log(`Gross Pay: `, grossPay)
        console.log(`Uncle Sam's Share: `, taxAmount)
        console.log(`Net Pay: `, netPay)

        //alert(`Gross Pay: $`, grossPay, `Uncle Sam's Share: $`, taxAmount, `Net Pay: $`, netPay)

        alert(`Gross Pay: $ ${grossPay.toFixed(2)} \n Uncle Sam's Share: $ ${taxAmount.toFixed(2)} \n Net Pay: $ ${netPay.toFixed(2)}`)

       




    </script>
